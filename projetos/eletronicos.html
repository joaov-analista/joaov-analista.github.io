Contexto do Projeto

Uma loja de eletrônicos hipotética apresentava quedas sazonais no faturamento.

O objetivo foi construir um pipeline completo: extrair os dados transacionais do MySQL, modelar um Star Schema otimizado para análise dimensional e criar um dashboard executivo com análise YoY que revelasse onde a receita estava sendo perdida.

O resultado mostrou que o gap de R$ +1M estava concentrado no mês de abril, impactado o faturamento — informação que não estava visível nos relatórios operacionais.

-- 1. Criação das dimensões (Star Schema)
CREATE TABLE dim_produto (
  id_produto    INT PRIMARY KEY AUTO_INCREMENT,
  nome_produto  VARCHAR(150),
  categoria     VARCHAR(80),
  marca         VARCHAR(80),
  preco_custo   DECIMAL(10,2),
  preco_venda   DECIMAL(10,2)
);

CREATE TABLE dim_tempo (
  id_tempo    INT PRIMARY KEY,   -- formato YYYYMMDD
  data        DATE,
  ano         SMALLINT,
  trimestre   TINYINT,
  mes         TINYINT,
  nome_mes    VARCHAR(15),
  semana      TINYINT,
  dia_semana  VARCHAR(15)
);

-- 2. Tabela fato de vendas
CREATE TABLE fato_vendas (
  id_venda    INT PRIMARY KEY AUTO_INCREMENT,
  id_tempo    INT,
  id_produto  INT,
  id_loja     INT,
  id_cliente  INT,
  quantidade  INT,
  receita     DECIMAL(10,2),
  custo       DECIMAL(10,2),
  lucro       DECIMAL(10,2),
  FOREIGN KEY (id_tempo)   REFERENCES dim_tempo(id_tempo),
  FOREIGN KEY (id_produto) REFERENCES dim_produto(id_produto)
);


-- 3. Análise YoY — receita por categoria
WITH receita_anual AS (
  SELECT
    p.categoria,
    t.ano,
    SUM(f.receita)                                      AS receita_total
  FROM fato_vendas f
  JOIN dim_produto p ON f.id_produto = p.id_produto
  JOIN dim_tempo   t ON f.id_tempo   = t.id_tempo
  WHERE t.ano IN (2024, 2025)
  GROUP BY p.categoria, t.ano
),
pivot AS (
  SELECT
    categoria,
    SUM(CASE WHEN ano = 2024 THEN receita_total END)    AS receita_2024,
    SUM(CASE WHEN ano = 2025 THEN receita_total END)    AS receita_2025
  FROM receita_anual
  GROUP BY categoria
)
SELECT
  categoria,
  ROUND(receita_2024, 2)                                AS receita_2024,
  ROUND(receita_2025, 2)                                AS receita_2025,
  ROUND(receita_2025 - receita_2024, 2)                 AS gap_receita,
  ROUND((receita_2025 - receita_2024)
        / receita_2024 * 100, 1)                        AS variacao_pct
FROM pivot
ORDER BY gap_receita ASC;


-- 4. Gap concentrado por trimestre e categoria
SELECT
  t.trimestre,
  p.categoria,
  ROUND(SUM(CASE WHEN t.ano = 2024 THEN f.receita END), 2) AS receita_2024,
  ROUND(SUM(CASE WHEN t.ano = 2025 THEN f.receita END), 2) AS receita_2025,
  ROUND(
    SUM(CASE WHEN t.ano = 2025 THEN f.receita END) -
    SUM(CASE WHEN t.ano = 2024 THEN f.receita END), 2)     AS gap
FROM fato_vendas f
JOIN dim_produto p ON f.id_produto = p.id_produto
JOIN dim_tempo   t ON f.id_tempo   = t.id_tempo
WHERE t.ano IN (2024, 2025)
GROUP BY t.trimestre, p.categoria
HAVING gap < 0
ORDER BY gap ASC
LIMIT 10;

import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.ticker as mtick
import seaborn as sns

sns.set_theme(style='whitegrid')
plt.rcParams['figure.dpi'] = 120

# ── 1. Carregamento dos dados ───────────────────
df = pd.read_csv('vendas_eletronicos.csv', parse_dates=['data_venda'])
df['ano']       = df['data_venda'].dt.year
df['mes']       = df['data_venda'].dt.month
df['trimestre'] = df['data_venda'].dt.quarter

print(f"Shape: {df.shape}")
print(df.dtypes)
print("\nValores nulos:\n", df.isnull().sum())

# ── 2. Receita YoY por categoria ────────────────
receita = (
    df.groupby(['categoria', 'ano'])['receita']
      .sum()
      .unstack('ano')
      .fillna(0)
      .assign(
          gap=lambda x: x[2025] - x[2024],
          variacao_pct=lambda x: (x[2025] - x[2024]) / x[2024] * 100
      )
      .sort_values('gap')
)
print("\nReceita por categoria:\n", receita.round(2))

# ── 3. Visualização: gap YoY por categoria ──────
fig, ax = plt.subplots(figsize=(9, 5))
cores = ['#EF4444' if v < 0 else '#10B981' for v in receita['gap']]
bars  = ax.barh(receita.index, receita['gap'], color=cores)
ax.axvline(0, color='#E2E8F0', linewidth=1)
ax.set_title('Gap de Receita YoY por Categoria (2024 vs 2025)',
             fontsize=13, fontweight='bold')
ax.set_xlabel('Gap de Receita (R$)')
for b in bars:
    w = b.get_width()
    ax.text(w + (500 if w >= 0 else -500),
            b.get_y() + b.get_height() / 2,
            f'R$ {w:,.0f}', va='center',
            ha='left' if w >= 0 else 'right', fontsize=9)
plt.tight_layout()
plt.savefig('outputs/gap_yoy_categoria.png', bbox_inches='tight')
plt.show()

# ── 4. Evolução mensal — 2024 vs 2025 ──────────
mensal = (
    df.groupby(['ano', 'mes'])['receita']
      .sum()
      .unstack('ano')
      .fillna(0)
)
fig, ax = plt.subplots(figsize=(11, 4))
ax.plot(mensal.index, mensal[2024], marker='o',
        label='2024', color='#93c5fd', linewidth=2)
ax.plot(mensal.index, mensal[2025], marker='o',
        label='2025', color='#2563EB', linewidth=2)
ax.fill_between(mensal.index, mensal[2024], mensal[2025],
                where=mensal[2025] < mensal[2024],
                alpha=0.12, color='#EF4444', label='Gap negativo')
ax.yaxis.set_major_formatter(
    mtick.FuncFormatter(lambda x, _: f'R$ {x/1000:.0f}K'))
ax.set_title('Evolução Mensal de Receita — 2024 vs 2025',
             fontsize=13, fontweight='bold')
ax.set_xlabel('Mês')
ax.legend()
plt.tight_layout()
plt.savefig('outputs/evolucao_mensal.png', bbox_inches='tight')
plt.show()

Insights do Projeto

Smartphones concentram 58% do gap
De R$ 42K de gap total, R$ 24,4K foram gerados apenas na categoria Smartphones — impactada por ruptura de estoque e mudança de mix de modelos em Q4.
Queda de 18% YoY concentrada no Q4
Os trimestres Q1, Q2 e Q3 apresentaram desempenho estável ou positivo. A queda total foi puxada pelo Q4, onde a receita recuou 18% em relação ao mesmo período do ano anterior.
Região Nordeste com maior potencial
Apesar de representar apenas 22% da receita, a região Nordeste foi a única com crescimento YoY positivo (+7%), indicando oportunidade de expansão não explorada.
